# leo-abigen

ABI-to-TypeScript binding generator for [Leo](https://leo-lang.org/) / Aleo smart contracts.
Analogous to `go-ethereum/abigen` for the Aleo ecosystem.

Given a contract's `abi.json` (produced by `leo build`), generates a fully typed TypeScript client with:

- **Transition builders** — type-safe transaction specs, no manual input serialization
- **Mapping readers** — pure HTTP reads, no WASM
- **Record scanners** — typed filtering of private wallet records

---

## End-to-end example

### 1. Start with a Leo contract

```leo
program fairswap_dex_v3.aleo {

    struct PoolState {
        token_a:     field,
        token_b:     field,
        reserve_a:   u128,
        reserve_b:   u128,
        lp_supply:   u128,
        fee_bps:     u16,
    }

    record LpToken {
        owner:    address,
        pool_key: field,
        amount:   u128,
    }

    mapping pools: field => PoolState;
    mapping lp_balances: field => u128;

    fn create_pool(
        public token_x: field,
        public token_y: field,
        public fee_bps: u16,
    ) -> Final { ... }

    fn swap(
        public token_in_id:  field,
        public token_out_id: field,
        public amount_in:    u128,
        public min_out:      u128,
        public recipient:    address,
    ) -> Final { ... }

    fn add_liquidity_private(
        record_a:          token_registry.aleo/Token,
        record_b:          token_registry.aleo/Token,
        public lp_to_mint: u128,
        public min_lp:     u128,
        public fee_bps:    u16,
        public recipient:  address,
    ) -> (LpToken, Final) { ... }
}
```

### 2. Generate the client

`leo build` produces `build/abi.json`. Feed it to the CLI:

```bash
leo-abigen --abi build/abi.json --out src/contracts/
# → src/contracts/fairswap-dex-v3.ts
```

The filename is derived from the program id: `fairswap_dex_v3.aleo` → `fairswap-dex-v3.ts`.

### 3. Generated output

The generated file contains everything needed to interact with the contract — no manual ABI imports:

```typescript
// auto-generated by leo-abigen — do not edit
// source: fairswap_dex_v3.aleo

import { createAbigen, createRecordScanner }  from "@fairdrop/leo-abigen";
import type { AbiRecord, TransitionHandle }   from "@fairdrop/leo-abigen";
import type { Address, Field, U128, WalletRecord } from "@fairdrop/types/primitives";

// ABI embedded — consumers never manage abi.json manually
export const _abi = JSON.parse('{ ... }') as Parameters<typeof createAbigen>[0];

// ── Structs ─────────────────────────────────────────────────────────

export interface PoolState {
  token_a: Field;
  token_b: Field;
  reserve_a: U128;
  reserve_b: U128;
  lp_supply: U128;
  fee_bps: number;
}

// ── Records ─────────────────────────────────────────────────────────

export interface LpTokenRecord {
  owner: Address;
  pool_key: Field;
  amount: U128;
  spent:   boolean;
  _record: string;   // plaintext — pass as transition input
}

export function scanLpTokenRecords(entries: WalletRecord[]): LpTokenRecord[] { ... }

// ── Transition args ─────────────────────────────────────────────────

export interface CreatePoolArgs {
  token_x: Field;
  token_y: Field;
  fee_bps: number;
}

export interface SwapArgs {
  token_in_id: Field;
  token_out_id: Field;
  amount_in: U128;
  min_out: U128;
  recipient: Address;
}

export interface AddLiquidityPrivateArgs {
  record_a: string | Record<string, unknown>;   // private record input
  record_b: string | Record<string, unknown>;
  lp_to_mint: U128;
  min_lp: U128;
  fee_bps: number;
  recipient: Address;
}

// ── Client interface ────────────────────────────────────────────────

export interface FairswapDexV3 {
  // Transitions — each is callable + has .build()
  createPool: TransitionHandle<CreatePoolArgs>;
  swap: TransitionHandle<SwapArgs>;
  addLiquidityPrivate: TransitionHandle<AddLiquidityPrivateArgs>;

  // Mapping reads — pure HTTP, no WASM
  pools(key: string): Promise<PoolState | null>;
  lpBalances(key: string): Promise<U128 | null>;

  // Record scanners
  records: {
    lpTokens(entries: WalletRecord[]): LpTokenRecord[];
  };
}

// ── Factory ─────────────────────────────────────────────────────────

export function createFairswapDexV3(
  config?: Parameters<typeof createAbigen>[1]
): FairswapDexV3 { ... }
```

### 4. Use it

**Builder mode** — build a transaction spec, execute it yourself:

```typescript
import { createFairswapDexV3 } from "./contracts/fairswap-dex-v3";
import { fromAleoClient } from "@fairdrop/leo-abigen/adapters";

const client = createFairswapDexV3({
  fetchMapping: fromAleoClient(networkClient),
});

// Build a transaction spec — pass to wallet adapter
const spec = client.swap.build({
  token_in_id:  "1field"   as Field,
  token_out_id: "2field"   as Field,
  amount_in:    "1000000"  as U128,
  min_out:      "990000"   as U128,
  recipient:    "aleo1..." as Address,
});
const { executeTransaction } = useWallet();
await executeTransaction(spec);
```

**Executor mode** — pass `executeTransaction` into the factory and call transitions directly:

```typescript
import { useMemo } from 'react';
import { useWallet } from '@provablehq/aleo-wallet-adaptor-react';
import { createFairswapDexV3 } from "./contracts/fairswap-dex-v3";

function useDexClient() {
  const { connected, executeTransaction } = useWallet();
  return useMemo(
    () => connected ? createFairswapDexV3({ executeTransaction }) : null,
    [connected, executeTransaction],
  );
}

// Then in a component:
const client = useDexClient();
const txId = await client.swap({ ... }); // executes immediately
```

**Mapping reads** — pure HTTP, no WASM:

```typescript
const pool = await client.pools("3443field");
console.log(pool?.reserve_a);  // U128 branded string
```

**Record scanning** — typed filtering of private wallet records:

```typescript
const lpTokens = client.records.lpTokens(walletRecords);
lpTokens.forEach(lp => console.log(lp.pool_key, lp.amount));
```

## Installation

```bash
npm install @fairdrop/leo-abigen @fairdrop/types
```

> `@provablehq/aleo-types` is a peer dependency provided automatically by any Provable wallet adapter.

## CLI reference

```bash
# Auto-derive filename from program id (recommended)
leo-abigen --abi build/abi.json --out src/generated/

# Explicit output filename
leo-abigen --abi build/abi.json --out src/generated/my-contract.ts

# Print to stdout
leo-abigen --abi build/abi.json
```

### Turbo integration

```json
// turbo.json
{
  "tasks": {
    "codegen": {
      "inputs":    ["build/abi.json"],
      "outputs":   ["src/generated/**"],
      "dependsOn": ["leo-abigen#build"]
    }
  }
}
```

```json
// your-contract/package.json
{
  "scripts": {
    "build":   "leo build",
    "codegen": "leo-abigen --abi build/abi.json --out src/generated/"
  }
}
```

## API

### `createXxx(config?)` — generated factory

Each generated file exports a typed factory named after the contract:

```typescript
const client = createFairswapDexV3({
  fetchMapping:       fromAleoClient(networkClient), // for mapping reads
  executeTransaction: wallet.executeTransaction,     // for executor mode
  fee:                500_000,                       // microcredits, default 300_000
  privateFee:         false,
});
```

### `createAbigen(abi, config)`

Creates a dynamic client object directly. Use when you have the ABI at hand and don't need the generated factory.

| Config option | Type | Description |
|---|---|---|
| `fetchMapping` | `MappingFetcher` | For mapping reads. Provide via `fromAleoClient()`. |
| `executeTransaction` | `(spec) => Promise<{transactionId}>` | For executor mode. |
| `fee` | `number` | Default fee in microcredits. Default: `300_000`. |
| `privateFee` | `boolean` | Pay fee from private balance. Default: `false`. |

### `fromAleoClient(client)`

Adapts an `AleoNetworkClient` (from `@provablehq/sdk`) to `MappingFetcher`. Converts thrown errors (key not found) to `null`.

### `TransitionHandle<A>`

Every generated transition is a `TransitionHandle<A>` — both callable and has a `.build()` property:

```typescript
// Executor mode — calls executeTransaction directly
const txId = await client.swap(args);

// Builder mode — returns TransactionOptions for the caller to execute
const spec = client.swap.build(args);
```

### `createRecordScanner(record, structs)`

Creates a typed scanner for a specific record type. Used internally by the generated code.

## Type mapping

| Leo type | TypeScript | Serialized (wire) form |
|---|---|---|
| `field` | `Field` (`Brand<string, 'Field'>`) | `"3443field"` |
| `u128` | `U128` (`Brand<string, 'U128'>`) | `"1000000u128"` |
| `u64` | `U64` (`Brand<string, 'U64'>`) | `"42u64"` |
| `u32` / `u16` / `u8` | `number` | `"100u32"` |
| `address` | `Address` (`Brand<string, 'Address'>`) | identity |
| `bool` | `boolean` | `"true"` / `"false"` |
| `signature` | `string` | identity |
| `[T; N]` | `T[]` | `"[val1, val2, ...]"` |
| Struct | Generated interface | `"{ field: value, ... }"` |
| Record input | `string \| Record<string, unknown>` | passed through to wallet |

## Architecture

```
src/
  runtime/          ← browser-safe, no WASM, published to npm
    abi.ts          ← ABI JSON schema types
    dispatch.ts     ← serialize / deserialize Leo values
    mapping.ts      ← createMappingReader
    transition.ts   ← TransitionHandle<A>, createTransitionBuilder
    records.ts      ← createRecordScanner
    adapters.ts     ← fromAleoClient adapter
    client.ts       ← createAbigen
  codegen/          ← Node.js CLI, build-time only
    index.ts        ← generateTypes(abi) + CLI entry
    structs.ts      ← emit struct interfaces
    records.ts      ← emit record interfaces + scanners
    transitions.ts  ← emit Args interfaces
    mappings.ts     ← emit mapping reader declarations
    client.ts       ← emit typed client interface + factory function
    utils.ts        ← toCamelCase, toPascalCase, type mapping helpers
  index.ts          ← public API re-exports
```
